cmake_minimum_required(VERSION 3.26)

project(CoralDAQ)

set( ${PROJECT_NAME}_VERSION_MAJOR 0 )
set( ${PROJECT_NAME}_VERSION_MINOR 1 )
set( ${PROJECT_NAME}_VERSION_PATCH 0 )

find_package(DATEMONITORING CONFIG REQUIRED)
find_package(EXPAT REQUIRED)

include_directories(src
                    ${DATEMONITORING_INCLUDE_DIRS}
                    ${CMAKE_CURRENT_BINARY_DIR})

set(USE_DATE 1)
set(USE_XROOTD 0)
set(USE_RFIO 0)

configure_file(src/config.h.cmake ${CMAKE_CURRENT_BINARY_DIR}/config.h @ONLY)

file(GLOB DAQSOURCES src/*.cc)
file(GLOB DAQHEADERS src/*.h)
list(APPEND DAQHEADERS ${CMAKE_CURRENT_BINARY_DIR}/config.h)

add_library(coral-ddd SHARED ${DAQSOURCES})

target_link_libraries(coral-ddd ${DATEMONITORING_LIBRARIES}
                              ${EXPAT_LIBRARIES})

install(TARGETS coral-ddd DESTINATION ${CMAKE_INSTALL_PREFIX}/lib64)

install(FILES ${DAQHEADERS} DESTINATION "${CMAKE_INSTALL_PREFIX}/include/na64/coral-ddd")
